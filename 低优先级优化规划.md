## 二、A2 — Bindings 细粒度缓存

### 现状问题
- 当前使用全量数组缓存 (`bindingsCache: any[]`)
- 30 秒 TTL 过期后全量失效
- 任何写操作都调用 `invalidateBindingsCache()` 清空全部缓存
- 高并发场景下缓存命中率低

### 优化目标
- 按 tagCode 细粒度缓存，单条 60 秒 TTL
- 写操作只失效受影响的 tagCode
- 查询时自动清理过期条目
- 支持按 tagCode 查询单个绑定关系

### 设计方案

```typescript
// 缓存结构: Map<tagCode, { entry, expireAt }>
const bindingsCache = new Map<string, {
  entry: TagBindingEntry;
  expireAt: number;
}>();

const CACHE_TTL = 60_000; // 60 秒

// 查询: 先查缓存，过期则从 API 拉取并更新
async function callTagBindings(): Promise<TagBindingEntry[]> {
  const now = Date.now();
  
  // 清理过期条目
  for (const [tagCode, cached] of bindingsCache) {
    if (cached.expireAt < now) bindingsCache.delete(tagCode);
  }
  
  // 如果缓存为空，全量拉取
  if (bindingsCache.size === 0) {
    const fresh = await fetchAllBindings();
    for (const entry of fresh) {
      bindingsCache.set(entry.tagCode, { entry, expireAt: now + CACHE_TTL });
    }
  }
  
  return Array.from(bindingsCache.values()).map(c => c.entry);
}

// 单条查询: 只查指定 tagCode
async function callTagBindingByCode(tagCode: string): Promise<TagBindingEntry | null> {
  const cached = bindingsCache.get(tagCode);
  if (cached && cached.expireAt > Date.now()) return cached.entry;
  // 未命中则全量拉取并更新缓存
  await callTagBindings();
  return bindingsCache.get(tagCode)?.entry ?? null;
}

// 失效: 只失效指定 tagCode
function invalidateBindingByTagCode(tagCode: string) {
  bindingsCache.delete(tagCode);
}

// 写操作后: 只失效受影响的标签
// 例: bind_tag_to_person → invalidateBindingByTagCode(newTagCode)
```

## 三、M1 — 指标持久化

### 现状问题
- 当前 metrics 存储在 `const recentCalls: MetricEntry[] = []`（内存）
- streamable-http 模式下，每次请求可能创建新的 Server 实例
- 指标在请求间不持久，每次 health_check 只看到当前请求的指标
- 无法追踪历史趋势和累积统计

### 优化目标
- 指标持久化到文件 (`data/metrics.json`)
- 跨请求累积统计
- 支持滚动窗口（最近 24 小时或 10000 条）
- health_check 返回真实累积指标

### 设计方案

```typescript
import { readFileSync, writeFileSync, existsSync } from 'fs';
import { join, dirname } from 'path';
import { fileURLToPath } from 'url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const METRICS_FILE = join(__dirname, '..', '..', 'data', 'metrics.json');

interface MetricsStore {
  entries: MetricEntry[];
  totalCalls: number;
  totalErrors: number;
  startedAt: string;
  lastFlush: string;
}

// 从文件加载
function loadMetrics(): MetricsStore {
  if (existsSync(METRICS_FILE)) {
    try {
      return JSON.parse(readFileSync(METRICS_FILE, 'utf-8'));
    } catch { return defaultStore(); }
  }
  return defaultStore();
}

function defaultStore(): MetricsStore {
  return { entries: [], totalCalls: 0, totalErrors: 0, 
           startedAt: new Date().toISOString(), lastFlush: '' };
}

// 异步刷盘（不阻塞工具调用）
async function flushMetrics(store: MetricsStore) {
  try {
    if (!existsSync(join(__dirname, '..', '..', 'data'))) {
      // auto-create if needed
    }
    writeFileSync(METRICS_FILE, JSON.stringify(store, null, 2));
    store.lastFlush = new Date().toISOString();
  } catch (err) { /* silent fail - metrics are best-effort */ }
}

// 收集指标（持久化版）
export function collectMetricsPersistent(toolName: string, fn) {
  return async (...args) => {
    const startTime = performance.now();
    let success = true;
    let error: string | undefined;
    try {
      return await fn(...args);
    } catch (err: any) {
      success = false; error = err.message; throw err;
    } finally {
      const duration = performance.now() - startTime;
      const store = loadMetrics();
      store.entries.push({ tool: toolName, duration, success, error, timestamp: new Date().toISOString() });
      store.totalCalls++;
      if (!success) store.totalErrors++;
      // 滚动窗口: 保留最近 10000 条
      if (store.entries.length > 10000) {
        store.entries = store.entries.slice(-10000);
      }
      // 异步刷盘（不阻塞响应）
      flushMetrics(store).catch(() => {});
    }
  };
}
```
